### 2.1 百炼 RAG 工具

保留阿里云百炼的 RAG 能力，作为知识库检索工具：

```python
from langchain.tools import tool
import requests

@tool
def bailian_rag(query: str) -> str:
    """
    从百炼知识库检索矿业项目信息。

    包含内容：
    - 项目档案（地质报告、可研报告、尽调材料）
    - 行业标准（JORC、NI 43-101、中国矿业权评估准则）
    - 专业知识（选矿工艺、采矿方法、矿业经济）

    使用场景：
    - 查询特定项目的详细信息
    - 检索行业标准和规范
    - 获取专业技术资料

    参数：
        query: 检索问题，如 "贵州磷矿项目的品位数据"

    返回：
        检索到的相关内容
    """
    # 百炼 API 配置
    BAILIAN_APP_ID = os.getenv("BAILIAN_APP_ID")
    BAILIAN_API_KEY = os.getenv("BAILIAN_API_KEY")

    response = requests.post(
        f"https://bailian.aliyuncs.com/v1/apps/{BAILIAN_APP_ID}/completion",
        headers={
            "Authorization": f"Bearer {BAILIAN_API_KEY}",
            "Content-Type": "application/json"
        },
        json={
            "input": {"prompt": query},
            "parameters": {
                "top_k": 5,  # 返回前5个相关片段
                "return_sources": True
            }
        }
    )

    result = response.json()
    if result.get("success"):
        return result["output"]["text"]
    else:
        return f"检索失败: {result.get('message', '未知错误')}"
```

### 2.2 联网搜索工具

使用 Tavily API 搜索外部信息：

```python
from langchain.tools import tool
from tavily import TavilyClient

@tool
def web_search(query: str, search_depth: str = "basic") -> str:
    """
    搜索互联网获取最新信息。

    使用场景：
    - 查询大宗商品实时价格
    - 搜索最新政策法规变化
    - 获取行业新闻和动态
    - 搜集竞品项目公开资料

    参数：
        query: 搜索关键词，如 "2026年铜价走势"
        search_depth: 搜索深度，"basic"（快速）或 "advanced"（深入）

    返回：
        搜索结果摘要
    """
    client = TavilyClient(api_key=os.getenv("TAVILY_API_KEY"))

    response = client.search(
        query=query,
        search_depth=search_depth,
        include_answer=True,
        max_results=5
    )

    # 格式化输出
    results = []
    if response.get("answer"):
        results.append(f"摘要: {response['answer']}\n")

    for item in response.get("results", []):
        results.append(f"- {item['title']}: {item['content'][:200]}...")

    return "\n".join(results)
```

### 3.2 IRR 计算工具

```python
from langchain.tools import tool

@tool
def calculate_irr(
    initial_investment: float,
    annual_cashflows: list,
    discount_rate: float = 0.10
) -> dict:
    """
    计算矿业项目的投资回报指标。

    参数：
        initial_investment: 初始投资额（正数，单位：万元）
        annual_cashflows: 各年净现金流列表（单位：万元）
        discount_rate: 折现率（默认 10%）

    返回：
        包含 IRR、NPV、投资回收期的字典

    示例：
        calculate_irr(
            initial_investment=10000,
            annual_cashflows=[2000, 2500, 3000, 3500, 4000],
            discount_rate=0.10
        )
    """
    # 构建完整现金流（第0年为负的初始投资）
    cashflows = [-initial_investment] + annual_cashflows

    # 计算 IRR（二分法）
    def npv_at_rate(rate):
        return sum(cf / (1 + rate) ** t for t, cf in enumerate(cashflows))

    lo, hi = -0.99, 10.0
    for _ in range(1000):
        mid = (lo + hi) / 2
        npv = npv_at_rate(mid)
        if abs(npv) < 1e-6:
            break
        if npv > 0:
            lo = mid
        else:
            hi = mid
    irr = mid

    # 计算 NPV
    npv = npv_at_rate(discount_rate)

    # 计算投资回收期
    cumsum = 0
    payback_period = None
    for t, cf in enumerate(cashflows):
        cumsum += cf
        if cumsum >= 0 and payback_period is None:
            # 线性插值计算精确回收期
            if t > 0:
                prev_cumsum = cumsum - cf
                payback_period = t - 1 + (-prev_cumsum) / cf
            else:
                payback_period = 0

    return {
        "IRR": f"{irr * 100:.2f}%",
        "NPV": f"{npv:,.0f} 万元",
        "投资回收期": f"{payback_period:.1f} 年" if payback_period else "超出计算期限",
        "折现率": f"{discount_rate * 100:.0f}%",
        "总投资": f"{initial_investment:,.0f} 万元",
        "项目周期": f"{len(annual_cashflows)} 年"
    }
```

### 3.3 敏感性分析工具

```python
@tool
def sensitivity_analysis(
    base_irr: float,
    base_npv: float,
    variable: str,
    changes: list = [-0.2, -0.1, 0, 0.1, 0.2],
    impact_factor: float = 1.0
) -> str:
    """
    对关键变量进行敏感性分析。

    参数：
        base_irr: 基准 IRR（如 0.15 表示 15%）
        base_npv: 基准 NPV（万元）
        variable: 变量名称（如 "铜价"、"品位"、"开采成本"）
        changes: 变化幅度列表（默认 ±20%, ±10%, 0）
        impact_factor: 影响系数（变量变化1%对IRR的影响）

    返回：
        敏感性分析表格

    示例：
        sensitivity_analysis(
            base_irr=0.15,
            base_npv=5000,
            variable="铜价",
            changes=[-0.2, -0.1, 0, 0.1, 0.2],
            impact_factor=0.8
        )
    """
    results = []
    results.append(f"## {variable} 敏感性分析\n")
    results.append(f"| {variable}变化 | IRR | NPV (万元) | 变化幅度 |")
    results.append("|---------|-----|----------|---------|")

    for change in changes:
        new_irr = base_irr + change * impact_factor * base_irr
        new_npv = base_npv * (1 + change * impact_factor)
        irr_change = (new_irr - base_irr) / base_irr * 100 if base_irr != 0 else 0

        results.append(
            f"| {change:+.0%} | {new_irr:.1%} | {new_npv:,.0f} | {irr_change:+.1f}% |"
        )

    results.append(f"\n**结论**: {variable}每变化10%，IRR约变化{impact_factor*10:.1f}%")

    return "\n".join(results)
```

## 五、人机交互工具

### 5.1 主动询问工具

```python
from langchain.tools import tool
from langgraph.types import interrupt

@tool
def ask_user(question: str, options: list = None) -> str:
    """
    向用户询问信息或确认。

    参数：
        question: 要问的问题
        options: 可选的选项列表，如 ["选项A", "选项B"]

    使用场景：
        - 关键数据在知识库中找不到
        - 需要用户确认评估方向
        - 发现重大风险需要提示
        - 评估参数需要校准

    示例：
        ask_user("请问这个磷矿的 P2O5 品位是多少？")
        ask_user("您说的'那个铜矿'是指哪个项目？", ["云南铜矿", "西藏铜矿", "其他"])
    """
    # interrupt 会暂停 Agent，等待用户输入后继续
    response = interrupt({
        "type": "question",
        "question": question,
        "options": options or []
    })

    return response
```

### 5.2 确认操作工具

```python
@tool
def confirm_action(action: str, reason: str) -> bool:
    """
    执行重要操作前请求用户确认。

    参数：
        action: 要执行的操作描述
        reason: 为什么需要确认

    返回：
        用户是否确认（True/False）

    示例：
        confirm_action(
            action="将评估结论写入项目档案",
            reason="这将更新甘肃镍矿项目的投资建议"
        )
    """
    response = interrupt({
        "type": "confirmation",
        "action": action,
        "reason": reason
    })

    return response.lower() in ["yes", "确认", "同意", "好的", "可以"]
```

### 7.1 Docstring 是关键

工具的 docstring 决定了模型何时、如何使用这个工具：

```python
# ❌ 不好的 docstring
@tool
def search(q):
    """搜索"""
    pass

# ✅ 好的 docstring
@tool
def web_search(query: str) -> str:

    返回：
        搜索结果摘要
    """

```
