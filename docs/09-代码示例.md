```python
        "model": "gpt-4o",
        "env_key": "OPENAI_API_KEY"
    }
}

def get_llm(model_name: str = "deepseek"):
    """获取 LLM 实例"""
    config = MODEL_CONFIG[model_name]
    api_key = os.getenv(config["env_key"])

    if not api_key:
        raise ValueError(f"Missing API key: {config['env_key']}")

    if config.get("provider") == "anthropic":
        return ChatAnthropic(
            model=config["model"],
            api_key=api_key
        )
    else:
        return ChatOpenAI(
            base_url=config["base_url"],
            model=config["model"],
            api_key=api_key
        )
```

```python
from langchain.tools import tool
import requests
import os

@tool
def bailian_rag(query: str) -> str:
    """
    从百炼知识库检索矿业项目信息。

    包含内容：
    - 项目档案（地质报告、可研报告、尽调材料）
    - 行业标准（JORC、NI 43-101、中国矿业权评估准则）
    - 专业知识（选矿工艺、采矿方法、矿业经济）

    参数：
        query: 检索问题，如 "贵州磷矿项目的品位数据"

    返回：
        检索到的相关内容
    """
    BAILIAN_APP_ID = os.getenv("BAILIAN_APP_ID")
    BAILIAN_API_KEY = os.getenv("BAILIAN_API_KEY")

    try:
        response = requests.post(
            f"https://bailian.aliyuncs.com/v1/apps/{BAILIAN_APP_ID}/completion",
            headers={
                "Authorization": f"Bearer {BAILIAN_API_KEY}",
                "Content-Type": "application/json"
            },
            json={
                "input": {"prompt": query},
                "parameters": {"top_k": 5}
            },
            timeout=30
        )

        result = response.json()
        if result.get("success"):
            return result["output"]["text"]
        else:
            return f"检索失败: {result.get('message', '未知错误')}"

    except Exception as e:
        return f"知识库连接失败: {str(e)}"
```

```python
from langchain.tools import tool
from tavily import TavilyClient
import os

@tool
def web_search(query: str, search_depth: str = "basic") -> str:
    """
    搜索互联网获取最新信息。

    使用场景：
    - 查询大宗商品实时价格
    - 搜索最新政策法规变化
    - 获取行业新闻和动态

    参数：
        query: 搜索关键词
        search_depth: "basic"（快速）或 "advanced"（深入）

    返回：
        搜索结果摘要
    """
    try:
        client = TavilyClient(api_key=os.getenv("TAVILY_API_KEY"))

        response = client.search(
            query=query,
            search_depth=search_depth,
            include_answer=True,
            max_results=5
        )

        results = []
        if response.get("answer"):
            results.append(f"**摘要**: {response['answer']}\n")

        for item in response.get("results", []):
            results.append(f"- {item['title']}: {item['content'][:200]}...")

        return "\n".join(results) if results else "未找到相关结果"

    except Exception as e:
        return f"搜索失败: {str(e)}"
```

```python
# src/tools/calculate.py
from langchain.tools import tool

@tool
def calculate_irr(
    initial_investment: float,
    annual_cashflows: list,
    discount_rate: float = 0.10
) -> dict:
    """
    计算矿业项目的投资回报指标。

    参数：
        initial_investment: 初始投资额（正数，单位：万元）
        annual_cashflows: 各年净现金流列表（单位：万元）
        discount_rate: 折现率（默认 10%）

    返回：
        包含 IRR、NPV、投资回收期的字典

    示例：
        calculate_irr(10000, [2000, 2500, 3000, 3500, 4000])
    """
    # 构建完整现金流
    cashflows = [-initial_investment] + annual_cashflows

    # 计算 IRR（二分法）
    def npv_at_rate(rate):
        return sum(cf / (1 + rate) ** t for t, cf in enumerate(cashflows))

    lo, hi = -0.99, 10.0
    for _ in range(1000):
        mid = (lo + hi) / 2
        npv = npv_at_rate(mid)
        if abs(npv) < 1e-6:
            break
        if npv > 0:
            lo = mid
        else:
            hi = mid
    irr = mid

    # 计算 NPV
    npv = npv_at_rate(discount_rate)

    # 计算投资回收期
    cumsum = 0
    payback_period = None
    for t, cf in enumerate(cashflows):
        cumsum += cf
        if cumsum >= 0 and payback_period is None:
            if t > 0:
                prev_cumsum = cumsum - cf
                payback_period = t - 1 + (-prev_cumsum) / cf if cf != 0 else t
            else:
                payback_period = 0

    return {
        "IRR": f"{irr * 100:.2f}%",
        "NPV": f"{npv:,.0f} 万元",
        "投资回收期": f"{payback_period:.1f} 年" if payback_period else "超出计算期限",
        "折现率": f"{discount_rate * 100:.0f}%",
        "总投资": f"{initial_investment:,.0f} 万元",
        "项目周期": f"{len(annual_cashflows)} 年"
    }


@tool
def sensitivity_analysis(
    base_params: dict,
    variable: str,
    changes: list = None
) -> str:
    """
    对关键变量进行敏感性分析。

    参数：
        base_params: 基准参数 {"initial_investment": 10000, "annual_cashflows": [...]}
        variable: 变量名称（"price"、"grade"、"cost"）
        changes: 变化幅度列表（默认 [-20%, -10%, 0, +10%, +20%]）

    返回：
        敏感性分析表格
    """
    if changes is None:
        changes = [-0.2, -0.1, 0, 0.1, 0.2]

    results = [f"## {variable} 敏感性分析\n"]
    results.append(f"| 变化幅度 | IRR | NPV (万元) |")
    results.append("|---------|-----|-----------|")

    base_cashflows = base_params["annual_cashflows"]
    initial = base_params["initial_investment"]

    for change in changes:
        # 调整现金流
        adjusted_cashflows = [cf * (1 + change) for cf in base_cashflows]
        result = calculate_irr.invoke({
            "initial_investment": initial,
            "annual_cashflows": adjusted_cashflows
        })

        results.append(f"| {change:+.0%} | {result['IRR']} | {result['NPV']} |")

    return "\n".join(results)
```

```python
# src/tools/interaction.py
from langchain.tools import tool
from langgraph.types import interrupt

@tool
def ask_user(question: str, options: list = None) -> str:
    """
    向用户询问信息或确认。

    参数：
        question: 要问的问题
        options: 可选的选项列表

    使用场景：
        - 关键数据在知识库中找不到
        - 需要用户确认评估方向
        - 发现重大风险需要提示
    """
    response = interrupt({
        "type": "question",
        "question": question,
        "options": options or []
    })
    return response
```

```python
# src/agent/main.py
from deepagents import create_deep_agent
from deepagents.backends import CompositeBackend, StateBackend, StoreBackend
from deepagents.middleware import HumanInTheLoopMiddleware
from langgraph.store.sqlite import SqliteStore
from langgraph.checkpoint.memory import MemorySaver

from .config import get_llm
from .prompts import MINING_SYSTEM_PROMPT
from ..tools.rag import bailian_rag
from ..tools.search import web_search
from ..tools.calculate import calculate_irr, sensitivity_analysis
from ..tools.interaction import ask_user


class MiningAgent:
    """矿业投资智能体"""

    def __init__(self, model_name: str = "deepseek", db_path: str = "./data/memories.db"):
        self.model_name = model_name
        self.db_path = db_path

        # 初始化存储
        self.store = SqliteStore.from_path(db_path)
        self.store.setup()
        self.checkpointer = MemorySaver()

        # 初始化 LLM
        self.llm = get_llm(model_name)

        # 工具列表
        self.tools = [
            bailian_rag,
            web_search,
            calculate_irr,
            sensitivity_analysis,
            ask_user,
        ]

        # 子代理配置
        self.subagents = [
            {
                "name": "researcher",
                "description": "深度研究某个专题",
                "prompt": "你是矿业研究专家，擅长深度调研...",
                "tools": [web_search, bailian_rag],
            },
            {
                "name": "calculator",
                "description": "执行复杂财务计算",
                "prompt": "你是财务分析专家...",
                "tools": [calculate_irr, sensitivity_analysis],
            },
        ]

        # 创建 Agent
        self.agent = self._create_agent()

    def _make_backend(self, runtime):
        return CompositeBackend(
            default=StateBackend(runtime),
            routes={"/memories/": StoreBackend(runtime)}
        )

    def _create_agent(self):
        return create_deep_agent(
            model=self.llm,
            store=self.store,
            backend=self._make_backend,
            checkpointer=self.checkpointer,
            tools=self.tools,
            subagents=self.subagents,
            middleware=[
                HumanInTheLoopMiddleware(
                    interrupt_on={"ask_user": False}  # ask_user 工具自己处理中断
                )
            ],
            system_prompt=MINING_SYSTEM_PROMPT,
        )

    def chat(self, message: str, thread_id: str) -> dict:
        """发送消息"""
        config = {"configurable": {"thread_id": thread_id}}

        response = self.agent.invoke({
            "messages": [{"role": "user", "content": message}]
        }, config=config)

        return response

    def resume(self, user_input: str, thread_id: str) -> dict:
        """恢复中断的对话"""
        from langgraph.types import Command

        config = {"configurable": {"thread_id": thread_id}}

        response = self.agent.invoke(
            Command(resume=user_input),
            config=config
        )

        return response

    def switch_model(self, model_name: str):
        """切换模型"""
        self.model_name = model_name
        self.llm = get_llm(model_name)
        self.agent = self._create_agent()
        return f"已切换到 {model_name}"


# 便捷函数
def create_mining_agent(**kwargs):
    return MiningAgent(**kwargs)
```

```python
# run.py
from src.agent.main import create_mining_agent
import uuid

def main():
    print("正在启动矿业投资智能体...")
    agent = create_mining_agent(model_name="deepseek")
    print("智能体已就绪！")

    thread_id = str(uuid.uuid4())
    print(f"会话 ID: {thread_id}")
    print("-" * 50)

    while True:
        user_input = input("\n你: ").strip()

        if user_input.lower() in ["exit", "quit", "退出"]:
            print("再见！")
            break

        if user_input.startswith("/switch "):
            model_name = user_input.split(" ")[1]
            result = agent.switch_model(model_name)
            print(f"系统: {result}")
            continue

        response = agent.chat(user_input, thread_id)

        # 处理中断
        if isinstance(response, dict) and response.get("__interrupt__"):
            interrupt = response["__interrupt__"]
            print(f"\nAI: {interrupt['question']}")
            if interrupt.get("options"):
                for i, opt in enumerate(interrupt["options"]):
                    print(f"  {i+1}. {opt}")

            answer = input("你的回答: ").strip()
            response = agent.resume(answer, thread_id)

        # 显示响应
        if isinstance(response, dict):
            content = response.get("messages", [{}])[-1].get("content", str(response))
        else:
            content = str(response)

        print(f"\nAI: {content}")

if __name__ == "__main__":
    main()
```
